\section{Temporal Interview Profiling System (TIPS)}

\subsection{Automated Interview Analysis System}

\begin{itemize}
    \item Processes pre-recorded video interviews
    \item Extracts behavioral signals from audio and video
    \item Evaluates candidate responses using LLMs
    \item Generates time-evolving performance scores
    \item Compares answers against job descriptions
\end{itemize}

\clearpage

% ============================================================================
% SLIDE 2: ABSTRACT
% ============================================================================

\section{Abstract}

\begin{itemize}
    \item TIPS is an automated interview analysis system for video interviews
    \item Multi-stage backend pipeline processes recordings through 6 stages
    \item Extracts audio signals: transcription, pitch, energy, speech rate
    \item Analyzes video signals: face detection, head pose, gaze stability
    \item Uses Qwen2.5-3B LLM for semantic relevance scoring
\end{itemize}

\clearpage

% ============================================================================
% SLIDE 3: PROBLEM STATEMENT
% ============================================================================

\section{Problem Statement}

\begin{itemize}
    \item \textbf{Subjectivity}: Human interviewers have inherent biases and varying standards
    \item \textbf{Limited Analysis Depth}: Manual evaluation misses vocal and temporal patterns
    \item \textbf{Scalability Issues}: Time for evaluations grows linearly with candidate volume
    \item \textbf{Lack of Standardization}: Comparing candidates across interviewers is difficult
    \item \textbf{No Temporal Insights}: Traditional scoring gives only aggregate scores
\end{itemize}

\clearpage

\section{Motivation}

\begin{itemize}

    \item Organizations invest significant resources with inconsistent results

\section{System Architecture}

\subsection{Pipeline Flow}

\begin{enumerate}
    \item \textbf{Input}: Interview Recording + Job Description
    \item \textbf{Stage 0}: Timebase Establishment
    \item \textbf{Stage 1}: Parallel Signal Extraction (Audio/Video)
    \item \textbf{Stage 2}: Temporal Segmentation (Q\&A Pairing)
    \item \textbf{Stage 3}: Behavioral Metrics Computation
    \item \textbf{Stage 4-5}: LLM Semantic Scoring + Verdict
\end{enumerate}

\subsection{Components}
\begin{itemize}
    \item Interview UI Layer (WebRTC Recording)
    \item Backend Pipeline (6-Stage Processing)
    \item Dashboard (Visualization - Planned)
\end{itemize}

% ============================================================================
% SLIDE 6: METHODOLOGY - PIPELINE OVERVIEW
% ============================================================================

\section{Methodology}

\subsection{Pipeline Overview}

\begin{itemize}
    \item \textbf{Stage 0}: Establishes canonical timebase from video metadata
    \item \textbf{Stage 1}: Parallel extraction of audio features, video features, transcription
    \item \textbf{Stage 2}: Identifies speaking segments and pairs Q\&A
    \item \textbf{Stage 3}: Computes behavioral metrics (confidence, fluency, eye contact)
    \item \textbf{Stage 4-5}: LLM scoring and final verdict aggregation
\end{itemize}

\clearpage

% ============================================================================
% SLIDE 7: METHODOLOGY - KEY STAGES
% ============================================================================

\subsection{Key Processing Stages}

\begin{itemize}
    \item \textbf{Faster-Whisper}: Speech-to-text with word-level timestamps
    \item \textbf{WebRTC VAD}: Voice activity detection for speech segments
    \item \textbf{MediaPipe}: Face detection, landmark tracking, head pose estimation
    \item \textbf{Librosa}: Audio features (pitch, energy, speech rate)
    \item \textbf{Qwen2.5-3B}: 4-bit quantized LLM for semantic evaluation
\end{itemize}
